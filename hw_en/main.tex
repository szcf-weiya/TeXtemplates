\section{Prove $a^2 + b^2 \ge 2ab$}

\begin{solution}
  $$
  a^2 + b^2 - 2ab = (a - b)^2 \ge 0\,.
  $$
  More proofs see \textcite{kkkWhy2ab2017}.
\end{solution}
